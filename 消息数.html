（1）未读数量

with temp as

 (select m.subject, m.created_at, decode(m.msg_status, 0, 1, 0) unread

    from messages m

   where m.to_id = ?

     and m.is_receive_del = 0

     and m.msg_type = 0

  union all

  select k.subject, d.created_at, decode(d.msg_status, 0, 1, 0) unread

    from message_markets k

    left join message_market_userdetails d

      on k.id = d.subject_id

   where d.to_userid = ?

     and d.is_receive_del = 0

     and d.msg_type = 0)

select *

  from (select (select count(1) from temp where temp.unread = 1) unReadCount,

               temp.subject lastMsgContent,

               temp.created_at lastMsgDate

          from temp

         order by temp.created_at desc nulls last)

 where rownum = 1
