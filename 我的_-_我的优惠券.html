with categories as

 (select DISTINCT categoryids.id

    from auction_categories categoryids

   start with categoryids.id in (select auc.auction_category_id

                                   from auctions auc, USER_ORDERS uo

                                  where uo.auction_id = auc.id

                                    and uo.id = ?) 　　

  connect by categoryids.id = prior categoryids.parent_id)

select acu.id,

       acu.user_id,

       ac.name,

       ac.total_price,

       ac.favorable_price,

       acu.create_at,

       ac.start_at,

       ac.end_at,

       ac.type,

       ac.valid_time,

       ac.valid_type,

       acu.status,

       ac.auction_user_type

  from USER_ORDERS            o,

       auctions               a,

       auction_coupon         ac,

       auction_coupon_special acs,

       auction_coupon_user    acu,

       categories             c

 where o.AUCTION_ID = a.id

   and acs.type = 2

   and acs.coupon_id = ac.id

   and a.AUCTION_CATEGORY_ID in (SELECT * from categories)

   and acs.special_id = c.id

   and ac.status = 1

   and acu.coupon_id = ac.id

   and (ac.valid_type = 0 or ac.valid_type is null)

   and ac.START_AT <= sysdate

   and ac.END_AT > sysdate

   and to_char(a.auction_no) not in

       (select ace.auction_id

          from auction_coupon_extral ace

         where ace.coupon_id = ac.id)

   and acu.status = 0

   and ac.type != 5

   and o.id = ?

   and acu.user_id = ?
